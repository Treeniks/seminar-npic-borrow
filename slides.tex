    \AxiomC{\(\Gamma, A \vdash B\)}

    \AxiomC{\(\Delta \vdash A\)}
    \RightLabel{\(\rightarrow\)-E}

  \end{prooftree}

  \begin{prooftree}
    \AxiomC{\(\Gamma \vdash A\)}

    \BinaryInfC{\(\Gamma, \Delta \vdash C\)}

\subsection{Ownership and Move Semantics}

\begin{frame}[fragile]{Rust -- Ownership and Move Semantics}
  \centering

  \begin{lst}{rust}
    struct Box<T: Copy> { ... }
    impl<T: Copy> Box<T> {
      fn alloc(v: T) -> Self {}   fn load(b: Self) -> (Self, T) {}
      fn free(b: Self) {}         fn store(b: Self, v: T) -> Self {}
    }
  \end{lst}

  \begin{lst}{rust}
    let b = Box::alloc(1);
    let c = b; // value gets moved into y
    Box::store(b, 2); // $\text{\Lightning}$
  \end{lst}

  \pause

  \begin{itemize}
    \item Non-Copy values always have a single owner.
    \item Owner can change (move).
  \end{itemize}
\end{frame}

\subsection{Borrowing}

\begin{frame}[fragile]{Rust -- Borrowing}
  \begin{lst}{rust}
    struct Box<T: Copy> { ... }
    impl<T: Copy> Box<T> {
      fn alloc(v: T) -> Self {}   fn load(b: |\onslide<2->|&|\onslide<1->|Self) -> T {}
      fn free(b: Self) {}         fn store(b: |\onslide<2->|&mut |\onslide<1->|Self, v: T) {}
    }
  \end{lst}

  \onslide<3->

  \begin{lst}{rust}
    let x = Box::alloc(1);
    let v = Box::load(&x); // immutably borrows x
    Box::store(&mut x, 2); // mutably borrows x
  \end{lst}

  \onslide<4->

  For a given value, at any given time, there can only be one of:
  \begin{itemize}
    \item any number of immutable borrows (shared reference)
    \item a \textbf{single} mutable borrow (exclusive reference)
  \end{itemize}

  \onslide<1->
\end{frame}
