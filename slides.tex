\begin{frame}{Intuitionistic Logic}
  \centering
  \begin{bnf}
    \(A, B, C\) : Proposition ::= \(X\) : Variable | \(A \rightarrow B\) : Implication | \(A \land B\) : Conjunction
  \end{bnf}

  \pause

  \hrule

  \begin{prooftree}
    \AxiomC{}
    \RightLabel{Id}
    \UnaryInfC{\(A \vdash A\)}
    \DisplayProof\qquad
    \AxiomC{\(\Delta, \Gamma \vdash A\)}
    \RightLabel{Exchange}
    \UnaryInfC{\(\Gamma, \Delta \vdash A\)}
  \end{prooftree}

  \begin{prooftree}
    \AxiomC{\(\Gamma, A, A \vdash B\)}
    \RightLabel{Contraction}
    \UnaryInfC{\(\Gamma, A \vdash B\)}
    \DisplayProof\qquad
    \AxiomC{\(\Gamma \vdash B\)}
    \RightLabel{Weakening}
    \UnaryInfC{\(\Gamma, A \vdash B\)}
  \end{prooftree}

  \begin{prooftree}

    \RightLabel{\(\rightarrow\)-I}
    \UnaryInfC{\(\Gamma \vdash A \rightarrow B\)}
    \DisplayProof\qquad

    \AxiomC{\(\Delta \vdash A\)}
    \RightLabel{\(\rightarrow\)-E}
    \BinaryInfC{\(\Gamma, \Delta \vdash B\)}
  \end{prooftree}

    \BinaryInfC{\(\Gamma, \Delta \vdash C\)}
  \end{prooftree}

\subsection{Intuitionistic Types}

\begin{frame}{Intuitionistic Types}{Grammar of Typed Lambda Calculus}
  \centering
  \begin{bnf}
    \(s,t,u,v,w\) ::= \(x\) : Variable
    | \(\lambda x\ldotp s\) : Function Definition | \(s\ t\) : Function Application
    | \((u, v)\) : Tuple Definition | let \((x, y) = t\); \(w\) : Tuple Destructuring
  \end{bnf}

  \vspace{2em}

  \begin{bnf}
    \(\tau, \pi, \omega\) ::= \(T\) : Base Type
    | \(\tau \rightarrow \pi\) : Functions
    | \(\pi \times \omega\) : Tuple
  \end{bnf}
\end{frame}

\begin{frame}{Intuitionistic Types}{Intuitionistic Types -- Judgement Rules}
  \centering
  \begin{prooftree}
    \AxiomC{}
    \RightLabel{Id}
    \UnaryInfC{\(x : \tau \vdash x : \tau\)}
    \DisplayProof\qquad
    \AxiomC{\(\Delta, \Gamma \vdash t : \tau\)}
    \RightLabel{Exchange}
    \UnaryInfC{\(\Gamma, \Delta \vdash t: \tau\)}
  \end{prooftree}

  \begin{prooftree}
    \AxiomC{\(\Gamma, y : \tau, z : \tau \vdash u : \pi\)}
    \RightLabel{Contraction}
    \UnaryInfC{\(\Gamma, x : \tau \vdash u[x/y,x/z] : \pi\)}
    \DisplayProof\qquad
    \AxiomC{\(\Gamma \vdash s : \pi\)}
    \RightLabel{Weakening}
    \UnaryInfC{\(\Gamma, x : \tau \vdash s : \pi\)}
  \end{prooftree}

  \begin{prooftree}
    \AxiomC{\(\Gamma, x : \tau \vdash t : \pi\)}
    \RightLabel{\(\rightarrow\)-I}
    \UnaryInfC{\(\Gamma \vdash \lambda x\ldotp t : \tau \rightarrow \pi\)}
    \DisplayProof\qquad
    \AxiomC{\(\Gamma \vdash s : \tau \rightarrow \pi\)}
    \AxiomC{\(\Delta \vdash t : \tau\)}
    \RightLabel{\(\rightarrow\)-E}
    \BinaryInfC{\(\Gamma, \Delta \vdash s\ t : \pi\)}
  \end{prooftree}

  \begin{prooftree}
    \AxiomC{\(\Gamma \vdash u : \tau\)}
    \AxiomC{\(\Delta \vdash v : \pi\)}
    \RightLabel{\(\times\)-I}
    \BinaryInfC{\(\Gamma, \Delta \vdash (u, v) : \tau \times \pi\)}
    \DisplayProof\qquad
    \AxiomC{\(\Gamma \vdash t : \tau \times \pi\)}
    \AxiomC{\(\Delta, x : \tau, y : \pi \vdash w : \omega\)}
    \RightLabel{\(\times\)-E}
    \BinaryInfC{\(\Gamma, \Delta \vdash \text{ let } (x, y) = t; w : \omega\)}
  \end{prooftree}

  {\small The notation \(u[x/y,x/z]\) means to replace all occurrences of \(y\) and \(z\) in \(u\) by \(x\).}
\end{frame}

\subsection{Linear Types}

\begin{frame}[fragile]{Linear Types}{Allocation of Memory -- Use After Free}
  \centering

  \begin{align*}
    &\vdash \texttt{alloc} : \tau \rightarrow *\tau &
    &\vdash \texttt{load} : *\tau \rightarrow \tau \\
    &\vdash \texttt{free} : *\tau \rightarrow () &
    &\vdash \texttt{store} : *\tau \rightarrow \tau \rightarrow ()
  \end{align*}

  \pause

  \hrule

  \begin{lst}[width = 0.6\textwidth]{rust}
    let x = alloc(v);
    let _ = free(x);
    load(x) // accessing x after free
  \end{lst}
\end{frame}

\begin{frame}[fragile]{Linear Types}{Allocation of Memory -- Linear Types}
  \centering

  \begin{align*}
    &\vdash \texttt{alloc} : \tau \rightarrow \langle *\tau\rangle &
    &\vdash \texttt{load} : \langle *\tau\rangle \rightarrow \tau \\
    &\vdash \texttt{free} : \langle *\tau\rangle \rightarrow () &
    &\vdash \texttt{store} : \langle *\tau\rangle \rightarrow \tau \rightarrow ()
  \end{align*}

  \pause

  \hrule

  \begin{lst}[width = 0.6\textwidth]{rust}
    let x = alloc(v);
    let w = load(x); // x is consumed here
    free(x) // does not exist anymore
  \end{lst}
\end{frame}

\begin{frame}[fragile]{Linear Types}{Allocation of Memory -- Capability-Passing Style}
  \centering

  \begin{align*}
    &\vdash \texttt{alloc} : \tau \rightarrow \langle *\tau\rangle &
    &\vdash \texttt{load} : \langle *\tau\rangle \rightarrow \langle *\tau\rangle \times \tau \\
    &\vdash \texttt{free} : \langle *\tau\rangle \rightarrow () &
    &\vdash \texttt{store} : \langle *\tau\rangle \rightarrow \tau \rightarrow \langle *\tau\rangle
  \end{align*}

  \pause

  \hrule

  \begin{lst}[width = 0.6\textwidth]{rust}
    let x0 = alloc(v);
    let (x1, w) = load(x0);
    free(x1) // new x1 so fine
  \end{lst}
\end{frame}
